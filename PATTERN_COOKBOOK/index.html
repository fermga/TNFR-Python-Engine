    üîß Using the Cookbook API¶

    Programmatic access to all recipes is available through the TNFRCookbook class:

    from tnfr.recipes import TNFRCookbook

# Initialize cookbook
cookbook = TNFRCookbook()

# Get specific recipe
crisis_recipe = cookbook.get_recipe(
    domain="therapeutic",
    use_case="crisis_intervention"
)

print(f"Sequence: {crisis_recipe.sequence}")
print(f"Health: {crisis_recipe.health_metrics.overall_health:.3f}")

# List all recipes in a domain
therapeutic_recipes = cookbook.list_recipes(
    domain="therapeutic",
    min_health=0.75
)

# Search recipes by keyword
learning_recipes = cookbook.search_recipes("learning")

# Get recommendation based on context
recommended = cookbook.recommend_recipe(
    context="Need rapid team formation for crisis project",
    constraints={"max_length": 10, "min_health": 0.80}
)
